```java
public interface Person {
    String character();

    String nation();

    String name();

    int age();
}
```

```arduino
public class Asia implements Person {

    private String character = "skin color is yellow";
    private String nation;
    private String name;
    private int age;

    public Asia(String nation, String name, int age) {
        this.nation = nation;
        this.name = name;
        this.age = age;
    }

    @Override
    public String character() {
        return this.character;
    }

    @Override
    public String nation() {
        return this.nation;
    }

    @Override
    public String name() {
        return this.name;
    }

    @Override
    public int age() {
        return this.age;
    }
}
```

```sql
public class Korean extends Asia {

    public Korean(String nation, String name, int age) {
        super(nation, name, age);
    }

    public static void main(String[] args) {
        Korean k = new Korean("Korea", "Kim", 30);

        System.out.println("my nation is " + k.nation());
        System.out.println("my character is " + k.character());
        System.out.println("my name is " + k.name());
        System.out.println("my age is " + k.age());
    }
}
```
